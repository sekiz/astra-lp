**components/LanguageSwitcher.tsx** — minimal 'use client' island. Only this component uses client JS for the switcher — all other layout content is static Server Components:

```typescript
'use client'

import { useLocale, useTranslations } from 'next-intl'
import { useRouter, usePathname } from 'next-intl/navigation'

export function LanguageSwitcher() {
  const locale = useLocale()
  const t = useTranslations('Nav')
  const router = useRouter()
  const pathname = usePathname()

  const handleSwitch = () => {
    const nextLocale = locale === 'en' ? 'tr' : 'en'
    router.replace(pathname, { locale: nextLocale })
  }

  return (
    <button
      onClick={handleSwitch}
      aria-label={locale === 'en' ? t('switchLanguageLabel') : t('switchLanguageLabelAlt')}
      className="text-sm font-medium px-2 py-1 rounded border border-[var(--color-border)] hover:border-[var(--color-accent)] transition-colors"
    >
      {t('switchLanguage')}
    </button>
  )
}
```

Note on next-intl v3 navigation imports: In next-intl v3, `useRouter` and `usePathname` for locale switching are imported from `'next-intl/navigation'`, NOT from `'next-intl/client'`. If the import fails, try `import { useRouter, usePathname } from 'next-intl'` as fallback.
