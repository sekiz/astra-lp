Icon choices (Claude's discretion per CONTEXT.md):
- Card 1 (Linux Kernel): `Shield` — security, stability framing
- Card 2 (Open Source): `Code2` — open code, transparency
- Card 3 (Agent Orchestration): `Network` — orchestration topology

The grid uses `grid-cols-1 md:grid-cols-3` — single column on mobile, 3-column at md (768px) and above. Each card has `data-card` attribute for Playwright test selectors.

```typescript
import { getTranslations } from 'next-intl/server'
import { Shield, Code2, Network } from 'lucide-react'

type Props = {
  locale: string
}

export async function FeaturesSection({ locale }: Props) {
  const t = await getTranslations({ locale, namespace: 'Features' })

  const cards = [
    {
      icon: <Shield size={28} className="text-[var(--color-accent)] mb-4" aria-hidden="true" />,
      title: t('card1Title'),
      body: t('card1Body'),
    },
    {
      icon: <Code2 size={28} className="text-[var(--color-accent)] mb-4" aria-hidden="true" />,
      title: t('card2Title'),
      body: t('card2Body'),
    },
    {
      icon: <Network size={28} className="text-[var(--color-accent)] mb-4" aria-hidden="true" />,
      title: t('card3Title'),
      body: t('card3Body'),
    },
  ]

  return (
    <section
      data-section="features"
      className="px-6 py-24 max-w-6xl mx-auto w-full"
    >
      <h2 className="text-3xl md:text-4xl font-bold text-[var(--color-text-primary)] text-center mb-16">
        {t('sectionTitle')}
      </h2>
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        {cards.map((card, index) => (
          <div
            key={index}
            data-card
            className="p-6 rounded-xl border border-[var(--color-border)] bg-[var(--color-bg-surface)]"
          >
            {card.icon}
            <h3 className="text-lg font-semibold text-[var(--color-text-primary)] mb-3">
              {card.title}
            </h3>
            <p className="text-sm leading-relaxed text-[var(--color-text-secondary)]">
              {card.body}
            </p>
          </div>
        ))}
      </div>
    </section>
  )
}
```
